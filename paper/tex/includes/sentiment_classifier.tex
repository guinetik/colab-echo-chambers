  def sanitize_phrase(phrase):
      result = ""
      for word in phrase.split(" "):
          review = re.sub(r, ' ', word)
          review = review.lower()
          review = review.split()
          ps = PorterStemmer()
          # Stemming
          review = [ps.stem(word) for word in review if word not in stopwords_set]
          if review and len(review[0]) > 2:
              result += review[0] + " "
      return result.strip()
